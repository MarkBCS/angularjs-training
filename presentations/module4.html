Testing AngularJS applications

Why make automated tests for JavaScript UI?

  • SPAs contain complex UI logic
  • Javascript is not a compiled language. Dynamic typing introduces bugs easily at runtime.
  • AngularJs is designed to be easily testable
    • DOM manipulation is separated from the controllers
    • Dependency Injection allows mocking of external dependencies
  • Makes refactoring and new feature development easier and less likely to break existing features.
    • Time used for automated tests is paid back in the long run
  • High quality code always comes with automated tests
    • Less errors for the end user
    • Less manual regression testing
    • Makes developers happy and secure when refactoring or making changes to existing code.

How much automated tests for UI?

  • Test coverage should be practical
    • 100% coverage is not usually needed
    • No point to test for self-evident
  • Tests should cover core functionality and complex UI logic
  • If bug is discovered, a test should be written to prevent the same code from breaking again

AngularJS test methods

Unit tests

  • Test a single unit of code
  • Controllers, services, filters, directives
  • Mocks for injected external dependencies

Testing every controller may not be practical

Moving logic from controllers to services, directives and filters is good practice and makes unit testing simpler

End to end tests

  • Test the whole application stack
  • E2E testing framework emulates user actions through browser
  • Framework fills forms, clicks buttons etc and verifies that UI reacts correctly

End to end tests are the most important form of testing a web application

End to end tests take the most time to make but pay off in the long run

Where automated tests are run?

Developers should always run unit tests locally when making UI changes. End to end tests should be run every time any changes are made.

CI system should run automated tests every time changes are pushed to repository

Unit tests in more detail

How to setup unit testing in AngularJS

  • Unit tests are run with a Node.js app Karma
  • Tests can be written with almost any JS testing library. Jasmine, Mocha and QUnit are supported out of the box by Karma.

Structure of Jasmine tests:

describe('componentName', function() {

    var injectedComponent;

    beforeEach(module('requiredModule'));

    beforeEach(inject(_injectedComponent_) {
        injectedComponent = _injectedComponent_;
    })

    it('Expected behaviour', function(){
        expect(injectedComponent.methodThatReturnsBoolean()).toBe(true);
    })
});

Testing AngularJS components with no dependencies

Let's write a unit test for a filter:

angular.module('app').filter('userFilter', function($filter) {
    return function(users, city){
        return $filter('filter')(users, {city : city});
    }
});

userFilterTest.js

describe('userFilter', function() {
    var userFilter;
    var user1 = {name : 'Tapio', city  : 'Tampere'};
    var user2 = {name : 'Marko', city : 'Tampere'};
    var user3 = {name : 'Kari', city  : 'Helsinki'};
    var user4 = {name : 'Kalle', city : 'Turku'};
    var users = [user1, user2, user3, user4];
    beforeEach(module('app'));
    beforeEach(inject(function($filter){
        userFilter = $filter('userFilter');
    }));
    it('Will filter users by city (Tampere)', function() {
        expect(userFilter(users, 'Tampere')).toEqual([user1, user2]);
    });
    it('Will filter users by city (Helsinki)', function() {
        expect(userFilter(users, 'Helsinki')).toEqual([user3]);
    });
    it('Will filter users by city (Turku)', function() {
        expect(userFilter(users, 'Turku')).toEqual([user4]);
    });
});

Testing AngularJS controller with mocked depencencies

Let's write a test for a controller:

angular.module('app').controller('appController', function($scope, userDetailsService, userService) {
    $scope.userDetails = userDetailsService.get();
    $scope.users = userService.query();
});

appControllerTest.js

describe('appController', function() {
    var $controller, $rootScope, $q;

    beforeEach(module('app'));

    var userDetails = {
        username : 'markorautajoki',
        firstname : 'Marko',
        lastname : 'Rautajoki'
    };

    var user1 = {name : 'Tapio', city  : 'Tampere'};
    var user2 = {name : 'Marko', city : 'Tampere'};
    var user3 = {name : 'Kari', city  : 'Helsinki'};
    var user4 = {name : 'Kalle', city : 'Turku'};

    var users = [user1, user2, user3, user4];

    var userDetailsServiceMock = {
        get : function() {
            return userDetails;
        }
    }

    var userServiceMock = {
        query : function() {
            return users;
        }
    }

    beforeEach(inject(function(_$controller_, _$rootScope_, _$q_){
        $controller = _$controller_;
        $rootScope = _$rootScope_;
        $q = _$q_;
    }));

    it('Will fetch user details and users list', function() {
        var scope = $rootScope.$new();
        $controller('appController', {$scope : scope, userDetailsService : userDetailsServiceMock, userService : userServiceMock});
        expect(scope.userDetails).toEqual(userDetails);
        expect(scope.users).toEqual(users);
    });
});

Using $httpBackend to test $resource calls

Let's write a test for a factory that uses $resource

angular.module('app').factory('userDetailsService', function($resource) {
    return $resource('/api/userdetails');
});
describe('userDetailsService', function() {
    var $httpBackend, userDetailsService;

    var apiUrl = '/api/userdetails';
    var mockUserDetails = {username : 'markorautajoki', firstname : 'Marko', lastname : 'Rautajoki'};

    beforeEach(module('app'));

    beforeEach(inject(function(_$httpBackend_, _userDetailsService_) {
        $httpBackend = _$httpBackend_;
        $httpBackend.when('GET', apiUrl).respond(mockUserDetails);
        userDetailsService = _userDetailsService_;
    }));
    it('Should fetch user details', function() {
        $httpBackend.expectGET(apiUrl);

        userDetailsService.get().$promise.then(function(result) {
            expect(result.username).toEqual(mockUserDetails.username);
            expect(result.firstname).toEqual(mockUserDetails.firstname);
            expect(result.lastname).toEqual(mockUserDetails.lastname);
        });

        $httpBackend.flush();

    });
    afterEach(function() {
        $httpBackend.verifyNoOutstandingExpectation();
        $httpBackend.verifyNoOutstandingRequest();
    });
});

End to end tests in more details

How to setup end to end testing in AngularJS

  • Recommended way of running e2e tests in AngularJS is by a Node.js program Protractor.
  • Protractor uses webdriver to control browser. Tests use Jasmine syntax.

How protractor works

  • Element api is used to locate elements in DOM
  • Element api creates promises, that abstract away the asynchronous nature of AngularJS application and allow tests to be written (mostly) without callbacks.
  • Element api uses locators that locate elements for instance by css.
  • Browser api is used to navigate between urls
  • Protractor api documentation

Basic structure of protractor tests

describe('Some web ui functionality', function() {

    beforeEach(function(){
        browser.get('http://appurl');
    });

    it('Should do something', function() {
        element(by.css('button.somebutton')).click();

        element(by.css('input.someinput')).sendKeys('sometext');

        element(by.model('modelName')).sendKeys('sometext');

        element.all(by.repeater('item in items')).first().click();

        expect(element(by.css('.result-text')).getText()).toEqual('sometext');
    }
});

describe('frontpage', function(){
    it('should load the front page', function(){

        expect(element(by.cssContainingText('h1', 'Page title')).isPresent()).toEqual(true);
    });
